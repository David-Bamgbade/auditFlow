#!/usr/bin/env python3
"""
codebase-map.py - Complete OOP Codebase Mapper + Validator

Describes your entire codebase in object-oriented terms AND validates code against it.
Shows classes, constructors, methods, properties, fields, instances, inheritance,
abstract classes, interfaces - all from AST analysis.

Usage:
  python codebase-map.py generate              # Generate .code-map.json
  python codebase-map.py show                  # Show project overview
  python codebase-map.py show-oop              # Show all OOP details
  python codebase-map.py show-oop <file>       # Show OOP details for a file
  python codebase-map.py inspect <file>        # Show what's in a file
  python codebase-map.py validate <file>       # Validate a file against map
  python codebase-map.py verify                # Validate all files
  python codebase-map.py verify --strict       # Validate all files (with docstrings)
"""

import ast
import sys
import json
import hashlib
from pathlib import Path
from typing import Dict, List, Set, Optional, Any
from dataclasses import dataclass, field, asdict
from collections import defaultdict

# ============================================
# SKIP PATTERNS
# ============================================

SKIP_PATTERNS = [
    '__pycache__', 'venv', '.venv', 'env', '.env',
    'node_modules', '.git', '.github',
    'site-packages', 'dist-packages',
    '.mypy_cache', '.pytest_cache', '.ruff_cache',
    'build', 'dist',
    '*.pyc', '*.pyo',
    '*.egg-info',
    '.idea', '.vscode',
]

# ============================================
# DATA CLASSES
# ============================================

@dataclass
class MethodInfo:
    name: str
    line: int
    params: List[str]
    is_async: bool = False
    is_abstract: bool = False
    is_static: bool = False
    is_class: bool = False
    is_property: bool = False
    is_helper: bool = False
    is_magic: bool = False
    is_public: bool = True
    return_type: Optional[str] = None

@dataclass
class ObjectField:
    name: str
    line: int
    type: Optional[str]
    is_public: bool
    is_static: bool = False

@dataclass
class ObjectInstance:
    name: str
    class_name: str
    line: int
    args: List[str]
    is_public: bool

@dataclass
class ClassInfo:
    name: str
    line: int
    is_public: bool
    inherits_from: List[str]
    constructor: Optional[MethodInfo]
    object_methods: List[MethodInfo]
    helper_methods: List[MethodInfo]
    static_methods: List[MethodInfo]
    class_methods: List[MethodInfo]
    magic_methods: List[MethodInfo]
    object_properties: List[MethodInfo]
    object_fields: List[ObjectField]
    object_instances: List[ObjectInstance]
    is_abstract: bool
    is_interface: bool
    docstring: Optional[str]

@dataclass
class FunctionInfo:
    name: str
    line: int
    is_public: bool
    is_async: bool
    params: List[str]
    docstring: Optional[str]

@dataclass
class ConstantInfo:
    name: str
    line: int
    value: Optional[str]

@dataclass
class FileInfo:
    path: str
    classes: List[ClassInfo]
    functions: List[FunctionInfo]
    constants: List[ConstantInfo]
    imports: List[Dict]
    lines: int
    hash: str

@dataclass
class ProjectMap:
    files: Dict[str, FileInfo] = field(default_factory=dict)
    all_classes: List[ClassInfo] = field(default_factory=list)
    all_functions: List[FunctionInfo] = field(default_factory=list)
    all_constants: List[ConstantInfo] = field(default_factory=list)

# MAPPER
# ============================================

class CodeMapper(ast.NodeVisitor):
    def __init__(self, filepath: Path):
        self.filepath = filepath
        self.classes: List[ClassInfo] = []
        self.functions: List[FunctionInfo] = []
        self.constants: List[ConstantInfo] = []
        self.imports: List[Dict] = []
        self.lines = 0
        self.current_class: Optional[str] = None
        self.current_methods: List[MethodInfo] = []
        self.current_fields: List[ObjectField] = []
        self.current_instances: List[ObjectInstance] = []
        self.defined_classes: Set[str] = set()
        
    def map_file(self, code: str) -> FileInfo:
        self.lines = len(code.splitlines())
        try:
            tree = ast.parse(code, filename=str(self.filepath))
            # First pass: collect class names
            for node in ast.walk(tree):
                if isinstance(node, ast.ClassDef):
                    self.defined_classes.add(node.name)
            # Second pass: visit all nodes
            self.visit(tree)
        except SyntaxError:
            pass
        
        return FileInfo(
            path=str(self.filepath),
            classes=self.classes,
            functions=self.functions,
            constants=self.constants,
            imports=self.imports,
            lines=self.lines,
            hash=hashlib.md5(code.encode()).hexdigest()
        )
    
    def visit_ClassDef(self, node: ast.ClassDef):
        is_public = not node.name.startswith('_')
        
        # Get base classes
        inherits_from = []
        for base in node.bases:
            if isinstance(base, ast.Name):
                inherits_from.append(base.id)
            elif isinstance(base, ast.Attribute):
                inherits_from.append(base.attr)
        
        # Check if abstract
        is_abstract = False
        for child in node.body:
            if isinstance(child, ast.FunctionDef):
                for decorator in child.decorator_list:
                    if isinstance(decorator, ast.Name) and decorator.id == 'abstractmethod':
                        is_abstract = True
                        break
                if is_abstract:
                    break
        
        # Check if interface (only abstract methods, no implementation)
        is_interface = False
        if is_abstract:
            has_implementation = False
            for child in node.body:
                if isinstance(child, ast.FunctionDef):
                    is_abstract_method = any(
                        isinstance(d, ast.Name) and d.id == 'abstractmethod'
                        for d in child.decorator_list
                    )
                    if not is_abstract_method and len(child.body) > 0:
                        if not (len(child.body) == 1 and isinstance(child.body[0], ast.Pass)):
                            has_implementation = True
                            break
            is_interface = not has_implementation
        
        # Store current class context
        self.current_class = node.name
        self.current_methods = []
        self.current_fields = []
        self.current_instances = []
        
        # Collect methods and fields
        methods = []
        helper_methods = []
        static_methods = []
        class_methods = []
        magic_methods = []
        properties = []
        constructor = None
        
        for child in node.body:
            if isinstance(child, (ast.FunctionDef, ast.AsyncFunctionDef)):
                method_info = self._extract_method_info(child)
                
                if method_info.is_magic and method_info.name == '__init__':
                    constructor = method_info
                elif method_info.is_magic:
                    magic_methods.append(method_info)
                elif method_info.is_static:
                    static_methods.append(method_info)
                elif method_info.is_class:
                    class_methods.append(method_info)
                elif method_info.is_property:
                    properties.append(method_info)
                elif method_info.is_helper:
                    helper_methods.append(method_info)
                else:
                    methods.append(method_info)
                
                self.current_methods.append(method_info)
            
            elif isinstance(child, ast.Assign):
                self._extract_fields(child)
        
        # Create class info
        class_info = ClassInfo(
            name=node.name,
            line=node.lineno,
            is_public=is_public,
            inherits_from=inherits_from,
            constructor=constructor,
            object_methods=methods,
            helper_methods=helper_methods,
            static_methods=static_methods,
            class_methods=class_methods,
            magic_methods=magic_methods,
            object_properties=properties,
            object_fields=self.current_fields,
            object_instances=[],  # Will be populated during visit_Call
            is_abstract=is_abstract,
            is_interface=is_interface,
            docstring=ast.get_docstring(node)
        )
        
        self.classes.append(class_info)
        
        # Visit child nodes to find instances
        self.generic_visit(node)
        
        # Update instances after visiting body
        class_info.object_instances = self.current_instances
        
        self.current_class = None
        self.current_methods = []
        self.current_fields = []
        self.current_instances = []
    
    def _extract_method_info(self, node) -> MethodInfo:
        name = node.name
        is_magic = name.startswith('__') and name.endswith('__')
        is_helper = name.startswith('_') and not is_magic
        is_public = not name.startswith('_')
        is_static = False
        is_class = False
        is_property = False
        is_abstract = False
        is_async = isinstance(node, ast.AsyncFunctionDef)
        
        # Check decorators
        for decorator in node.decorator_list:
            if isinstance(decorator, ast.Name):
                if decorator.id == 'staticmethod':
                    is_static = True
                elif decorator.id == 'classmethod':
                    is_class = True
                elif decorator.id == 'property':
                    is_property = True
                elif decorator.id == 'abstractmethod':
                    is_abstract = True
        
        # Get parameters (skip self/cls)
        params = []
        for arg in node.args.args:
            if arg.arg not in ['self', 'cls']:
                params.append(arg.arg)
        
        return MethodInfo(
            name=name,
            line=node.lineno,
            params=params,
            is_async=is_async,
            is_abstract=is_abstract,
            is_static=is_static,
            is_class=is_class,
            is_property=is_property,
            is_helper=is_helper,
            is_magic=is_magic,
            is_public=is_public,
            return_type=None
        )
    
    def _extract_fields(self, node: ast.Assign):
        """Extract fields from assignments"""
        for target in node.targets:
            if isinstance(target, ast.Attribute) and isinstance(target.value, ast.Name):
                if target.value.id == 'self':
                    name = target.attr
                    is_public = not name.startswith('_')
                    
                    # Try to infer type
                    type_str = None
                    if isinstance(node.value, ast.Constant):
                        type_str = type(node.value.value).__name__
                    elif isinstance(node.value, ast.List):
                        type_str = 'List'
                    elif isinstance(node.value, ast.Dict):
                        type_str = 'Dict'
                    elif isinstance(node.value, ast.Set):
                        type_str = 'Set'
                    elif isinstance(node.value, ast.Tuple):
                        type_str = 'Tuple'
                    elif isinstance(node.value, ast.Call):
                        if isinstance(node.value.func, ast.Name):
                            type_str = node.value.func.id
                    
                    self.current_fields.append(ObjectField(
                        name=name,
                        line=node.lineno,
                        type=type_str,
                        is_public=is_public,
                        is_static=False
                    ))
    
    def visit_Call(self, node: ast.Call):
        """Track object instantiations"""
        if isinstance(node.func, ast.Name):
            class_name = node.func.id
            
            # Check if it's a class (defined in this file or known)
            if class_name in self.defined_classes:
                # Get arguments
                args = []
                for arg in node.args:
                    if isinstance(arg, ast.Constant):
                        args.append(repr(arg.value))
                    elif isinstance(arg, ast.Name):
                        args.append(arg.id)
                    elif isinstance(arg, ast.Constant) and isinstance(arg.value, str):
                        args.append(repr(arg.value))
                    else:
                        args.append('...')
                
                # Check if the call is being assigned to a variable
                parent = getattr(node, 'parent', None)
                if isinstance(parent, ast.Assign):
                    for target in parent.targets:
                        if isinstance(target, ast.Name):
                            self.current_instances.append(ObjectInstance(
                                name=target.id,
                                class_name=class_name,
                                line=node.lineno,
                                args=args,
                                is_public=not target.id.startswith('_')
                            ))
        
        self.generic_visit(node)
    
    def visit_FunctionDef(self, node: ast.FunctionDef):
        if self.current_class is None:
            # Top-level function (not in class)
            is_public = not node.name.startswith('_')
            params = [arg.arg for arg in node.args.args]
            self.functions.append(FunctionInfo(
                name=node.name,
                line=node.lineno,
                is_public=is_public,
                is_async=False,
                params=params,
                docstring=ast.get_docstring(node)
            ))
        self.generic_visit(node)
    
    def visit_AsyncFunctionDef(self, node: ast.AsyncFunctionDef):
        if self.current_class is None:
            is_public = not node.name.startswith('_')
            params = [arg.arg for arg in node.args.args]
            self.functions.append(FunctionInfo(
                name=node.name,
                line=node.lineno,
                is_public=is_public,
                is_async=True,
                params=params,
                docstring=ast.get_docstring(node)
            ))
        self.generic_visit(node)
    
    def visit_Assign(self, node: ast.Assign):
        """Extract top-level constants"""
        if self.current_class is None:
            for target in node.targets:
                if isinstance(target, ast.Name) and target.id.isupper():
                    value_str = None
                    if isinstance(node.value, ast.Constant):
                        value_str = str(node.value.value)
                    self.constants.append(ConstantInfo(
                        name=target.id,
                        line=node.lineno,
                        value=value_str
                    ))
        self.generic_visit(node)
    
    def visit_Import(self, node: ast.Import):
        for alias in node.names:
            self.imports.append({
                'source': alias.name,
                'alias': alias.asname or alias.name,
                'type': 'import',
                'line': node.lineno
            })
        self.generic_visit(node)
    
    def visit_ImportFrom(self, node: ast.ImportFrom):
        module = node.module or ''
        for alias in node.names:
            self.imports.append({
                'source': f"{module}.{alias.name}" if module else alias.name,
                'alias': alias.asname or alias.name,
                'type': 'import_from',
                'module': module,
                'line': node.lineno
            })
        self.generic_visit(node)

class Scanner:
    def __init__(self, root_path: str):
        self.root = Path(root_path).resolve()
        self.map = ProjectMap()
        
    def scan(self) -> ProjectMap:
        for py_file in self.root.rglob("*.py"):
            if self._should_skip(py_file):
                continue
            self._scan_file(py_file)
        
        # Build global lists
        for fileinfo in self.map.files.values():
            self.map.all_classes.extend(fileinfo.classes)
            self.map.all_functions.extend(fileinfo.functions)
            self.map.all_constants.extend(fileinfo.constants)
        
        return self.map
    
    def _should_skip(self, filepath: Path) -> bool:
        parts = filepath.parts
        for pattern in SKIP_PATTERNS:
            if pattern.startswith('*'):
                if pattern[1:] in filepath.name:
                    return True
            elif pattern in parts:
                return True
        return False
    
    def _scan_file(self, filepath: Path):
        try:
            code = filepath.read_text(encoding='utf-8')
            rel_path = str(filepath.relative_to(self.root))
            mapper = CodeMapper(filepath)
            fileinfo = mapper.map_file(code)
            fileinfo.path = rel_path
            self.map.files[rel_path] = fileinfo
        except Exception as e:
            print(f"Warning: Could not scan {filepath}: {e}")

def format_class_info(cls: ClassInfo, indent: str = "") -> str:
    """Format a class in OOP terms"""
    lines = []
    
    # Class header
    public_label = "Yes" if cls.is_public else "No"
    abstract_label = "Yes" if cls.is_abstract else "No"
    interface_label = "Yes" if cls.is_interface else "No"
    inherits = ", ".join(cls.inherits_from) if cls.inherits_from else "None"
    
    lines.append(f"{indent}📦 {cls.name} (class) - line {cls.line}")
    lines.append(f"{indent}   📤 Public: {public_label}")
    lines.append(f"{indent}   🔗 Inherits from: {inherits}")
    lines.append(f"{indent}   🎯 Abstract: {abstract_label}")
    lines.append(f"{indent}   📋 Interface: {interface_label}")
    
    # Constructor
    if cls.constructor:
        params = ", ".join([p for p in cls.constructor.params if p != 'self'])
        async_label = "async " if cls.constructor.is_async else ""
        lines.append(f"{indent}\n   🏗️ Constructor: {async_label}__init__({params}) -> line {cls.constructor.line}")
    else:
        lines.append(f"{indent}\n   🏗️ Constructor: None")
    
    # Object Methods
    if cls.object_methods:
        lines.append(f"{indent}\n   📋 Object Methods ({len(cls.object_methods)}):")
        for m in cls.object_methods:
            params = ", ".join(m.params)
            async_label = "async " if m.is_async else ""
            abstract_label = " [ABSTRACT]" if m.is_abstract else ""
            lines.append(f"{indent}     - {async_label}{m.name}({params}) -> line {m.line}{abstract_label}")
    else:
        lines.append(f"{indent}\n   📋 Object Methods: None")
    
    # Helper Methods
    if cls.helper_methods:
        lines.append(f"{indent}\n   📋 Helper Methods ({len(cls.helper_methods)}):")
        for m in cls.helper_methods:
            params = ", ".join(m.params)
            lines.append(f"{indent}     - {m.name}({params}) -> line {m.line}")
    
    # Static Methods
    if cls.static_methods:
        lines.append(f"{indent}\n   📋 Static Methods ({len(cls.static_methods)}):")
        for m in cls.static_methods:
            params = ", ".join(m.params)
            lines.append(f"{indent}     - {m.name}({params}) -> line {m.line}")
    
    # Class Methods
    if cls.class_methods:
        lines.append(f"{indent}\n   📋 Class Methods ({len(cls.class_methods)}):")
        for m in cls.class_methods:
            params = ", ".join(m.params)
            lines.append(f"{indent}     - {m.name}({params}) -> line {m.line}")
    
    # Magic Methods
    if cls.magic_methods:
        lines.append(f"{indent}\n   📋 Magic Methods ({len(cls.magic_methods)}):")
        for m in cls.magic_methods:
            params = ", ".join(m.params)
            lines.append(f"{indent}     - {m.name}({params}) -> line {m.line}")
    
    # Object Properties
    if cls.object_properties:
        lines.append(f"{indent}\n   📋 Object Properties ({len(cls.object_properties)}):")
        for p in cls.object_properties:
            lines.append(f"{indent}     - {p.name} -> line {p.line}")
    else:
        lines.append(f"{indent}\n   📋 Object Properties: None")
    
    # Object Fields
    if cls.object_fields:
        lines.append(f"{indent}\n   📦 Object Fields ({len(cls.object_fields)}):")
        for f in cls.object_fields:
            type_str = f" ({f.type})" if f.type else ""
            visibility = "public" if f.is_public else "private"
            lines.append(f"{indent}     - {f.name}{type_str} - {visibility}")
    else:
        lines.append(f"{indent}\n   📦 Object Fields: None")
    
    # Object Instances
    if cls.object_instances:
        lines.append(f"{indent}\n   🏗️ Object Instances Created ({len(cls.object_instances)}):")
        for inst in cls.object_instances:
            args_str = ", ".join(inst.args)
            lines.append(f"{indent}     - {inst.name} = {inst.class_name}({args_str}) -> line {inst.line}")
    else:
        lines.append(f"{indent}\n   🏗️ Object Instances: None")
    
    return "\n".join(lines)

class Validator:
    def __init__(self, map_data: ProjectMap, root_path: str, strict: bool = False):
        self.map = map_data
        self.root = Path(root_path).resolve()
        self.strict = strict
        self.issues: List[Dict] = []
        
        # Build lookup maps
        self.public_map = {}
        self.private_map = {}
        self.file_classes = {}
        self.dependency_graph = {}
        
        for filepath, fileinfo in self.map.files.items():
            self.public_map[filepath] = []
            self.private_map[filepath] = []
            self.file_classes[filepath] = []
            self.dependency_graph[filepath] = []
            
            # Classes
            for cls in fileinfo.classes:
                self.file_classes[filepath].append(cls.name)
                if cls.is_public:
                    self.public_map[filepath].append(cls.name)
                else:
                    self.private_map[filepath].append(cls.name)
                
                # Methods
                for method in cls.object_methods:
                    if method.is_public:
                        self.public_map[filepath].append(method.name)
                    else:
                        self.private_map[filepath].append(method.name)
                
                # Helper methods (usually private)
                for method in cls.helper_methods:
                    self.private_map[filepath].append(method.name)
            
            # Functions
            for func in fileinfo.functions:
                if func.is_public:
                    self.public_map[filepath].append(func.name)
                else:
                    self.private_map[filepath].append(func.name)
    
    def validate(self, filepath: str, code: str) -> List[Dict]:
        self.issues = []
        
        try:
            tree = ast.parse(code, filename=filepath)
        except SyntaxError as e:
            return [{'line': e.lineno, 'message': f"Syntax error: {e}", 'severity': 'ERROR'}]
        
        rel_path = str(Path(filepath).relative_to(self.root)) if Path(filepath).is_relative_to(self.root) else filepath
        
        # All checks
        self._check_private_imports(tree, rel_path)
        self._check_non_existent_imports(tree, rel_path)
        self._check_wildcard_imports(tree, rel_path)
        self._check_duplicate_imports(tree, rel_path)
        self._check_import_outside_root(tree, rel_path)
        self._check_import_in_function(tree, rel_path)
        self._check_test_import_in_prod(tree, rel_path)
        self._check_naming(tree, rel_path)
        self._check_function_length(tree, rel_path)
        self._check_parameters(tree, rel_path)
        self._check_docstrings(tree, rel_path)
        self._check_complexity(tree, rel_path)
        self._check_unused_vars(tree, rel_path)
        self._check_dead_code(tree, rel_path)
        
        return self.issues
    
    def _check_private_imports(self, tree: ast.AST, filepath: str):
        """Check: Importing private items"""
        for node in ast.walk(tree):
            if isinstance(node, ast.ImportFrom) and node.module:
                module_path = node.module.replace('.', '/') + '.py'
                if module_path in self.private_map:
                    for alias in node.names:
                        if alias.name in self.private_map.get(module_path, []):
                            self.issues.append({
                                'line': node.lineno,
                                'message': f"Importing private item '{alias.name}' from {node.module}",
                                'severity': 'ERROR',
                                'check': 'private_import'
                            })
    
    def _check_non_existent_imports(self, tree: ast.AST, filepath: str):
        """Check: Importing non-existent items"""
        for node in ast.walk(tree):
            if isinstance(node, ast.ImportFrom) and node.module:
                module_path = node.module.replace('.', '/') + '.py'
                public_items = self.public_map.get(module_path, [])
                private_items = self.private_map.get(module_path, [])
                all_items = public_items + private_items
                
                for alias in node.names:
                    if alias.name not in all_items and alias.name != '*':
                        self.issues.append({
                            'line': node.lineno,
                            'message': f"'{alias.name}' not found in {node.module}",
                            'severity': 'ERROR',
                            'check': 'non_existent_import'
                        })
    
    def _check_wildcard_imports(self, tree: ast.AST, filepath: str):
        """Check: Wildcard imports"""
        for node in ast.walk(tree):
            if isinstance(node, ast.ImportFrom):
                if any(alias.name == '*' for alias in node.names):
                    self.issues.append({
                        'line': node.lineno,
                        'message': f"Wildcard import from '{node.module}' - import what you need",
                        'severity': 'WARNING',
                        'check': 'wildcard_import'
                    })
    
    def _check_duplicate_imports(self, tree: ast.AST, filepath: str):
        """Check: Duplicate imports"""
        seen = set()
        for node in ast.walk(tree):
            if isinstance(node, ast.ImportFrom) and node.module:
                for alias in node.names:
                    key = f"{node.module}.{alias.name}"
                    if key in seen:
                        self.issues.append({
                            'line': node.lineno,
                            'message': f"Duplicate import of '{key}'",
                            'severity': 'INFO',
                            'check': 'duplicate_import'
                        })
                    seen.add(key)
    
    def _check_import_outside_root(self, tree: ast.AST, filepath: str):
        """Check: Import outside project root"""
        for node in ast.walk(tree):
            if isinstance(node, ast.ImportFrom) and node.module:
                if node.module.startswith('.'):
                    dots = 0
                    clean = node.module
                    while clean.startswith('.'):
                        dots += 1
                        clean = clean[1:]
                    
                    current = Path(filepath).parent
                    for _ in range(dots):
                        if current == self.root:
                            self.issues.append({
                                'line': node.lineno,
                                'message': f"Import '{node.module}' goes outside project root",
                                'severity': 'ERROR',
                                'check': 'import_outside_root'
                            })
                            break
                        current = current.parent
    
    def _check_import_in_function(self, tree: ast.AST, filepath: str):
        """Check: Import inside function"""
        for node in ast.walk(tree):
            if isinstance(node, ast.FunctionDef):
                for child in ast.walk(node):
                    if isinstance(child, (ast.Import, ast.ImportFrom)):
                        self.issues.append({
                            'line': child.lineno,
                            'message': f"Import inside function '{node.name}' - move to top",
                            'severity': 'WARNING',
                            'check': 'import_in_function'
                        })
    
    def _check_test_import_in_prod(self, tree: ast.AST, filepath: str):
        """Check: Production importing tests"""
        if 'test_' in filepath or 'tests/' in filepath:
            return
        
        for node in ast.walk(tree):
            if isinstance(node, ast.ImportFrom) and node.module:
                if 'test_' in node.module or 'tests/' in node.module:
                    self.issues.append({
                        'line': node.lineno,
                        'message': f"Production code imports from tests: '{node.module}'",
                        'severity': 'ERROR',
                        'check': 'test_import_in_prod'
                    })
    
    def _check_naming(self, tree: ast.AST, filepath: str):
        """Check: Naming conventions"""
        for node in ast.walk(tree):
            if isinstance(node, ast.ClassDef):
                if not node.name[0].isupper():
                    self.issues.append({
                        'line': node.lineno,
                        'message': f"Class '{node.name}' should be PascalCase",
                        'severity': 'INFO',
                        'check': 'naming_convention'
                    })
            elif isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
                if not node.name.startswith('_') and '_' not in node.name and not node.name[0].isupper():
                    self.issues.append({
                        'line': node.lineno,
                        'message': f"Function '{node.name}' should be snake_case",
                        'severity': 'INFO',
                        'check': 'naming_convention'
                    })
    
    def _check_function_length(self, tree: ast.AST, filepath: str):
        """Check: Function length"""
        for node in ast.walk(tree):
            if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
                length = node.end_lineno - node.lineno + 1 if hasattr(node, 'end_lineno') else 0
                if length > 50:
                    self.issues.append({
                        'line': node.lineno,
                        'message': f"Function '{node.name}' has {length} lines (suggest: max 50)",
                        'severity': 'WARNING',
                        'check': 'function_length'
                    })
    
    def _check_parameters(self, tree: ast.AST, filepath: str):
        """Check: Too many parameters"""
        for node in ast.walk(tree):
            if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
                count = len(node.args.args)
                if count > 5:
                    self.issues.append({
                        'line': node.lineno,
                        'message': f"Function '{node.name}' has {count} parameters (suggest: max 5)",
                        'severity': 'WARNING',
                        'check': 'too_many_parameters'
                    })
    
    def _check_docstrings(self, tree: ast.AST, filepath: str):
        """Check: Missing docstrings (skip private unless strict)"""
        if not self.strict:
            return
        
        for node in ast.walk(tree):
            if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef, ast.ClassDef)):
                if node.name.startswith('_'):
                    continue
                if 'test_' in filepath or 'tests/' in filepath:
                    continue
                if not ast.get_docstring(node):
                    kind = 'Function' if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)) else 'Class'
                    self.issues.append({
                        'line': node.lineno,
                        'message': f"{kind} '{node.name}' missing docstring",
                        'severity': 'INFO',
                        'check': 'missing_docstring'
                    })
    
    def _check_complexity(self, tree: ast.AST, filepath: str):
        """Check: High complexity"""
        for node in ast.walk(tree):
            if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
                complexity = 1
                for child in ast.walk(node):
                    if isinstance(child, (ast.If, ast.While, ast.For, ast.AsyncFor)):
                        complexity += 1
                    elif isinstance(child, (ast.And, ast.Or)):
                        complexity += 1
                    elif isinstance(child, ast.ExceptHandler):
                        complexity += 1
                if complexity > 15:
                    self.issues.append({
                        'line': node.lineno,
                        'message': f"Function '{node.name}' has complexity {complexity} (suggest: max 15)",
                        'severity': 'WARNING',
                        'check': 'high_complexity'
                    })
    
    def _check_unused_vars(self, tree: ast.AST, filepath: str):
        """Check: Unused variables with location details"""
        # Track assignments and their locations
        assigned = {}  # name -> (line, function_name)
        used = set()
        current_function = "global"
        
        # First pass: find all function definitions to track context
        for node in ast.walk(tree):
            if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
                # Visit function body to track assignments inside
                for child in ast.walk(node):
                    if isinstance(child, ast.Assign):
                        for target in child.targets:
                            if isinstance(target, ast.Name):
                                name = target.id
                                if not name.startswith('_'):
                                    assigned[name] = (child.lineno, node.name)
                    elif isinstance(child, ast.Name):
                        if isinstance(child.ctx, ast.Load):
                            used.add(child.id)
        
        # Second pass: track top-level assignments
        for node in ast.walk(tree):
            if isinstance(node, ast.Assign):
                # Skip if inside a function (already handled)
                parent = getattr(node, 'parent', None)
                if parent and isinstance(parent, (ast.FunctionDef, ast.AsyncFunctionDef)):
                    continue
                for target in node.targets:
                    if isinstance(target, ast.Name):
                        name = target.id
                        if not name.startswith('_'):
                            assigned[name] = (node.lineno, 'global')
            elif isinstance(node, ast.Name):
                if isinstance(node.ctx, ast.Load):
                    used.add(node.id)
        
        # Find unused variables
        for name, (line, func_name) in assigned.items():
            if name not in used and not name.startswith('_'):
                context = f"function '{func_name}'" if func_name != 'global' else "global scope"
                self.issues.append({
                    'line': line,
                    'message': f"Variable '{name}' assigned but never used in {context}",
                    'severity': 'WARNING',
                    'check': 'unused_variable'
                })
    
    def _check_dead_code(self, tree: ast.AST, filepath: str):
        """Check: Dead code after return with location"""
        for node in ast.walk(tree):
            if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
                found_return = False
                for child in node.body:
                    if isinstance(child, ast.Return):
                        found_return = True
                    elif found_return:
                        self.issues.append({
                            'line': child.lineno,
                            'message': f"Dead code after return in function '{node.name}' at line {child.lineno}",
                            'severity': 'WARNING',
                            'check': 'dead_code'
                        })
                        break

def generate_map(root_path: str):
    """Generate .code-map.json from the codebase"""
    print(f"📁 Mapping: {root_path}")
    scanner = Scanner(root_path)
    map_data = scanner.scan()
    
    map_file = Path(root_path) / '.code-map.json'
    with open(map_file, 'w') as f:
        data = {
            'files': {path: asdict(entry) for path, entry in map_data.files.items()},
        }
        json.dump(data, f, indent=2, default=str)
    
    total_classes = sum(len(f.classes) for f in map_data.files.values())
    total_functions = sum(len(f.functions) for f in map_data.files.values())
    total_constants = sum(len(f.constants) for f in map_data.files.values())
    
    print(f"✅ Mapped {len(map_data.files)} files")
    print(f"   📦 Classes: {total_classes}")
    print(f"   🔧 Functions: {total_functions}")
    print(f"   📊 Constants: {total_constants}")

def show_info(root_path: str):
    """Show project overview from the map"""
    map_file = Path(root_path) / '.code-map.json'
    if not map_file.exists():
        print("❌ Run 'generate' first")
        return
    
    with open(map_file) as f:
        data = json.load(f)
    
    total_classes = 0
    total_functions = 0
    total_constants = 0
    total_items = 0
    total_public = 0
    total_private = 0
    
    for entry in data['files'].values():
        total_classes += len(entry['classes'])
        total_functions += len(entry['functions'])
        total_constants += len(entry['constants'])
        total_items += len(entry['classes']) + len(entry['functions']) + len(entry['constants'])
        
        # Count public/private
        for cls in entry['classes']:
            if cls['is_public']:
                total_public += 1
            else:
                total_private += 1
    
    print(f"\n📋 Project: {Path(root_path).name}")
    print(f"   📁 Files: {len(data['files'])}")
    print(f"   📦 Total Items: {total_items}")
    print(f"   📦 Classes: {total_classes}")
    print(f"   🔧 Functions: {total_functions}")
    print(f"   📊 Constants: {total_constants}")
    print(f"   🔓 Public: {total_public}")
    print(f"   🔒 Private: {total_private}")
    
    print("\n📁 Files with most classes:")
    sorted_files = sorted(
        [(path, len(entry['classes'])) for path, entry in data['files'].items()],
        key=lambda x: x[1],
        reverse=True
    )[:5]
    for path, count in sorted_files:
        if count > 0:
            print(f"   {path}: {count} classes")

def show_oop_info(root_path: str, filepath: str = None):
    """Show OOP details for classes in the project or a specific file"""
    map_file = Path(root_path) / '.code-map.json'
    if not map_file.exists():
        print("❌ Run 'generate' first")
        return
    
    with open(map_file) as f:
        data = json.load(f)
    
    # Helper to convert dict back to dataclass
    def dict_to_classinfo(d):
        return ClassInfo(
            name=d['name'],
            line=d['line'],
            is_public=d['is_public'],
            inherits_from=d.get('inherits_from', []),
            constructor=MethodInfo(**d['constructor']) if d.get('constructor') else None,
            object_methods=[MethodInfo(**m) for m in d.get('object_methods', [])],
            helper_methods=[MethodInfo(**m) for m in d.get('helper_methods', [])],
            static_methods=[MethodInfo(**m) for m in d.get('static_methods', [])],
            class_methods=[MethodInfo(**m) for m in d.get('class_methods', [])],
            magic_methods=[MethodInfo(**m) for m in d.get('magic_methods', [])],
            object_properties=[MethodInfo(**m) for m in d.get('object_properties', [])],
            object_fields=[ObjectField(**f) for f in d.get('object_fields', [])],
            object_instances=[ObjectInstance(**i) for i in d.get('object_instances', [])],
            is_abstract=d.get('is_abstract', False),
            is_interface=d.get('is_interface', False),
            docstring=d.get('docstring')
        )
    
    if filepath:
        file_path = Path(filepath)
        rel_path = str(file_path.relative_to(root_path)) if file_path.is_relative_to(root_path) else str(file_path)
        if rel_path not in data['files']:
            print(f"❌ File not in map: {filepath}")
            return
        files_to_show = {rel_path: data['files'][rel_path]}
    else:
        files_to_show = data['files']
    
    print(f"\n📋 OOP Details:")
    
    for path, entry in files_to_show.items():
        if not entry['classes']:
            continue
        
        print(f"\n📁 {path}:")
        for cls_data in entry['classes']:
            cls = dict_to_classinfo(cls_data)
            print(format_class_info(cls, "  "))
            print()

                is_interface=cls_data.get('is_interface', False),

            ))
        
        map_data.files[path] = FileInfo(
            path=path,
            classes=classes,
            functions=[FunctionInfo(**f) for f in entry.get('functions', [])],
            constants=[ConstantInfo(**c) for c in entry.get('constants', [])],
            imports=entry.get('imports', []),
            lines=entry['lines'],
            hash=entry.get('hash', '')
        )
    
    file_path = Path(filepath)
    if file_path.is_dir():
        print(f"❌ '{filepath}' is a directory. Specify a file.")
        return
    
    if not file_path.exists():
        print(f"❌ File not found: {filepath}")
        return
    
    try:
        code = file_path.read_text(encoding='utf-8')
    except Exception as e:
        print(f"❌ Could not read file: {e}")
        return
    
    validator = Validator(map_data, root_path, strict=False)
    rel_path = str(file_path.relative_to(root_path)) if file_path.is_relative_to(root_path) else str(file_path)
    issues = validator.validate(rel_path, code)
    
    if issues:
        print(f"\n📋 Issues in {filepath}:")
        for issue in issues:
            line = f"Line {issue['line']}" if issue['line'] else "File"
            print(f"  [{issue['severity']}] {line}: {issue['message']}")
        print(f"\n📊 Total: {len(issues)} issues")
    else:
        print(f"✅ No issues in {filepath}")

def main():
    if len(sys.argv) < 2:
        print(__doc__)
        sys.exit(1)
    
    command = sys.argv[1]
    
    # Check for --strict flag
    strict = '--strict' in sys.argv
    
    if command == 'generate':
        root = sys.argv[2] if len(sys.argv) > 2 else '.'
        generate_map(root)
    
    elif command == 'show':
        root = sys.argv[2] if len(sys.argv) > 2 else '.'
        show_info(root)
    
    elif command == 'show-oop':
        root = sys.argv[2] if len(sys.argv) > 2 else '.'
        # Check if there's a file argument (not a flag)
        if len(sys.argv) >= 4 and not sys.argv[3].startswith('--'):
            show_oop_info(root, sys.argv[3])
        elif len(sys.argv) >= 3 and not sys.argv[2].startswith('--'):
            # If only one argument after command, treat as root
            show_oop_info(root)
        else:
            show_oop_info(root)
    
    elif command == 'inspect':
        if len(sys.argv) < 4:
            print("Usage: python codebase-map.py inspect <root> <file>")
            print("Example: python codebase-map.py inspect . reviewer.py")
            sys.exit(1)
        root = sys.argv[2]
        filepath = sys.argv[3]
        inspect_file(root, filepath)
    
    elif command == 'validate':
        if len(sys.argv) < 4:
            print("Usage: python codebase-map.py validate <root> <file>")
            print("Example: python codebase-map.py validate . test_slop.py")
            sys.exit(1)
        root = sys.argv[2]
        filepath = sys.argv[3]
        validate_file(root, filepath)
    
    elif command == 'verify':
        root = sys.argv[2] if len(sys.argv) > 2 else '.'
        verify_all(root, strict)
    
    else:
        print(f"Unknown: {command}")
        print(__doc__)

if __name__ == "__main__":
